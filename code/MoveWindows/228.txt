class Solution {
public:
    vector<string> summaryRanges(vector<int>& nums) {
        int len=nums.size();
		
		std::vector<string> ans;
		if(len==1){
			ans.push_back(std::to_string(nums[0]));
			return ans;
		}
		int i=0;
		while(i<len){
			int low=i;
			i++;
			while(i<len&&nums[i]==nums[i-1]+1){
				i++;
			}
			int high=i-1;
			string s=std::to_string(nums[low]);
			if(low<high){
				s+="->";
				s+=std::to_string(nums[high]);
			}
			ans.push_back(s);
		}
		return ans;
    }
};
