class Solution {
public:
    int climbStairs(int n) {
        int dp[50];
        dp[0]=0;
        dp[1]=1;
        dp[2]=2;
        for(int i=3;i<=n;i++){
            if(i-2>=0){
                dp[i]=dp[i-1]+dp[i-2];
            }else{
                dp[i]=dp[i-1]+1;
            }
        }
        return dp[n];
    }
};
